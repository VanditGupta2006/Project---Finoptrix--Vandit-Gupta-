\noindent
\textbf{FinOptix -- Stock Selection \& Portfolio Optimization} \hfill \textit{(Summer '25)}\\[2pt]
\textit{Finance \& Analytics Club, IIT Kanpur}

\vspace{6pt}

\renewcommand{\arraystretch}{1.3}
\noindent
\begin{tabular}{|>{\centering\arraybackslash}p{1.8cm}|p{13cm}|}
\hline

Objective &
\begin{tabitemize}
    \item To build a systematic investment strategy integrating \textbf{ML-based return prediction}, fundamental analysis, and \textbf{portfolio optimisation}, outperforming a market benchmark.
\end{tabitemize}\\
\hline

Approach &
\begin{tabitemize}
    \item Trained an \textbf{XGBoost Regressor} per stock on 15+ engineered features; selected features via PCA, Mutual Information, and Random Forest importance.
    \item Applied \textbf{Fama-French Three-Factor} regression for prior returns and blended with ML views using the \textbf{Black-Litterman} model, then optimised weights for maximum Sharpe Ratio.
\end{tabitemize}\\
\hline

Impact &
\begin{tabitemize}
    \item Fama-French + Black-Litterman portfolio achieved \textbf{46.42\% expected annual return}, outperforming plain MVO portfolios by over 10\%.
    \item Optimised portfolio delivered \textbf{$\sim$3$\times$ cumulative returns} over 2020--2025, consistently beating the equal-weight benchmark.
\end{tabitemize}\\
\hline

\end{tabular}
